#include<stdio.h>
#include<string.h>

void maiuscole(char [], char []);
void ordina(char []);

	maiuscole(s, s2);

void maiuscole(char s[], char s2[]){
	int i, j=0;
	for(i=0;s[i]!='\0';i++)
		if(s[i]>='A' && s[i]<='Z'){
			s2[j]=s[i];
			j++;
		}
	s2[j]='\0';
}

void ordina(char s2[]){
	int i, j;
	char min;
	for(i=1;s2[i]!='\0';i++){
		min=s2[i];
		j=i-1;
		while(min>s2[j] && j>=0){
			s2[j+1]=s2[j];
			j=j-1;
		}
		s2[j+1]=min;
	}
}
