Esercizio 8 | Puntatori e File
Dato un file si testo "prova.txt" riversare
il file su un secondo file "prova2.txt", ma al posto
della 'f' va scritta la stringa "effe".
es: caffettiera -> caeffeeffettiera
    #include<stdio.h>
        #include<stdlib.h>
        int main()
        {
            FILE *fp,*fpout;
            fp=fopen("prova.txt","r");
            fpout=fopen("output.txt","w");
            if (fp==NULL){
                perror("errore apertura file");
                exit(1);
            }
            char car,str[]="effe";
            while((car=fgetc(fp))!=EOF){	
                if(car=='f')
                    fputs(str,fpout);
                else
                    fputc(car,fpout);
            }
            fclose(fp);
            fclose(fpout);
            
            return 0;
        }
