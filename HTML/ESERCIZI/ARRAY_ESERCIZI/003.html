Esercizio 3 | Array
Caricare un vettore v con DIM=10 con numeri random a piacere
- inserire un numero k
- determinare l’elemento di v più prossimo a k e la relativa posizione
    #include<stdio.h>
        #include<math.h>
        #define DIM 10
        #define MAX 101
        
        int main()
        {
            int v[DIM],i,k,p=MAX,c=1,cf=0,s=0,ps=MAX;
            srand(time(NULL)); 
        
            printf("inserisci il numero: ");
            scanf("%d",&k);
            fflush(stdin);
            for(i=0;i<DIM;i++){ 
                v[i]=rand()%100+1;
                printf("%d  ",v[i]);
            }
            for(i=0;i<DIM;i++){
                s=k-v[i];	
                s=abs(s);
                if(s<ps){		
                    ps=s;	
                    p=v[i];
                    cf=c;	
                }
                c++;
            }
            if(p==MAX)
                printf("\n\nERRORE");
            else
                printf("\nl'elemento  di v  piu' prossimo a %d e' %d e si trova in posizione %d",k,p,cf);
            
            return 0;
        }
