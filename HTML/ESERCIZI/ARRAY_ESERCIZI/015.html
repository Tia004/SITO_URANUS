Caricare un vettore v di DIM=20 con i primi DIM numeri della successione di FIBONACCI.
Visualizzare un vettore v.

1 1 3 5 8 13 21 34 55 ...
(NB: esiste un altro numero palindromo nella successione, dopo il 55?)
    #include<stdio.h>
        #define DIM 10
        
        int main()
        {
            int v[DIM],i;
                
            v[0]=v[1]=1;
            for(i=2;i<DIM;i++){
                v[i]=v[i-1]+v[i-2];
            }
            for(i=0;i<DIM;i++){
                printf("%d\t",v[i]);
            }
            return 0;
        }
